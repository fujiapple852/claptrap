Parse boolean values.

# Example

### Spec

<Tabs syncKey="format">
  <TabItem label="TOML">
    ```toml
    name = "prog"

    [args]
    foo = { long = "foo", value-parser = ":bool:" }
    ```
  </TabItem>

  <TabItem label="YAML">
    ```yaml
    name: prog
    args:
      foo:
        long: foo
        value-parser: ':bool:'
    ```
  </TabItem>

  <TabItem label="JSON">
    ```json
    {
      "name": "prog",
      "args": {
        "foo": {
          "long": "foo",
          "value-parser": ":bool:"
        }
      }
    }
    ```
  </TabItem>
</Tabs>

### Input

```bash
prog --foo true
```

### Output

```bash
claptrap_foo='true'
```
